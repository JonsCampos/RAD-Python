CREATE TABLE Setor (
    setor_ID INT PRIMARY KEY AUTO_INCREMENT,
    nome_setor VARCHAR(50),
    desc_setor VARCHAR(255)
);

CREATE TABLE Funcionario (
    funcionario_ID INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(50),
    dt_nasc DATE,
    CPF VARCHAR(14),
    telefone VARCHAR(20),
    email VARCHAR(100),
    setor_ID INT,
    FOREIGN KEY (setor_ID) REFERENCES Setor(setor_ID)
);

CREATE TABLE Local (
    local_ID INT PRIMARY KEY AUTO_INCREMENT,
    nome_local VARCHAR(50)
);

CREATE TABLE Entrada (
    entrada_ID INT PRIMARY KEY AUTO_INCREMENT,
    data_hora_ent TIMESTAMP,
    funcionario_ID INT,
    local_ID INT,
    FOREIGN KEY (funcionario_ID) REFERENCES Funcionario(funcionario_ID),
    FOREIGN KEY (local_ID) REFERENCES Local(local_ID)
);

CREATE TABLE Saida (
    saida_ID INT PRIMARY KEY AUTO_INCREMENT,
    data_hora_sai TIMESTAMP,
    funcionario_ID INT,
    local_ID INT,
    FOREIGN KEY (funcionario_ID) REFERENCES Funcionario(funcionario_ID),
    FOREIGN KEY (local_ID) REFERENCES Local(local_ID)
);

// ENTRADA //
SELECT e.entrada_ID, f.nome, s.nome_setor, l.nome_local, TO_CHAR(e.data_hora_ent, 'DD/MM/YYYY HH24:MI') 
FROM Entrada e
JOIN Funcionario f ON e.funcionario_ID = f.funcionario_ID
JOIN Setor s ON f.setor_ID = s.setor_ID
JOIN Local l ON e.local_ID = l.local_ID
ORDER BY e.data_hora_ent DESC;


// SAÍDA //
SELECT sd.saida_ID, f.nome, st.nome_setor, l.nome_local, TO_CHAR(sd.data_hora_sai, 'DD/MM/YYYY HH24:MI') 
FROM Saida sd
JOIN Funcionario f ON sd.funcionario_ID = f.funcionario_ID
JOIN Setor st ON f.setor_ID = st.setor_ID
JOIN Local l ON sd.local_ID = l.local_ID
ORDER BY sd.data_hora_sai DESC;
